### 2.2 EmotionEngine — 情感计算引擎

```python
import re, json, time
from dataclasses import dataclass

class EmotionEngine:
    EMOTION_JSON_PATTERN = re.compile(
        r'\{[^{}]*"emotion"\s*:\s*\{[^}]+\}[^{}]*\}',
        re.DOTALL
    )

    def __init__(self):
        self._history: dict[str, list[EmotionState]] = {}
        self._last_active: dict[str, float] = {}

    def infer(self, reply: str, npc_name: str) -> EmotionState:
        """从 LLM 回复中解析情感"""
        self._last_active[npc_name] = time.time()

        match = self.EMOTION_JSON_PATTERN.search(reply)
        if not match:
            return self._default_state()

        try:
            data = json.loads(match.group())
            raw = data.get("emotion", {})
            emotion = EmotionState(
                primary=raw.get("primary", "neutral"),
                secondary=raw.get("secondary"),
                intensity=clamp(float(raw.get("intensity", 0.5)), 0.0, 1.0),
                valence=clamp(float(raw.get("valence", 0.0)), -1.0, 1.0),
                arousal=clamp(float(raw.get("arousal", 0.5)), 0.0, 1.0),
                dominance=clamp(float(raw.get("dominance", 0.5)), 0.0, 1.0),
                trend=raw.get("trend", "stable"),
                trigger_word=raw.get("trigger_word", ""),
            )
        except (json.JSONDecodeError, KeyError, ValueError):
            return self._default_state()

        # 校验 primary 合法性
        if emotion.primary not in [
            "happy","sad","angry","fearful","surprised",
            "disgusted","excited","worried","proud","neutral"
        ]:
            emotion.primary = "neutral"

        # 计算 trend
        history = self._history.setdefault(npc_name, [])
        if history:
            last = history[-1]
            if emotion.primary == last.primary:
                emotion.trend = (
                    "increasing" if emotion.intensity > last.intensity
                    else "decreasing"
                )
            else:
                emotion.trend = "stable"
        history.append(emotion)
        return emotion

    def decay(self, npc_name: str, seconds_idle: float) -> EmotionState:
        """静默衰减：超过60s无对话，情感归零"""
        if seconds_idle < 60:
            h = self._history.get(npc_name, [])
            return h[-1] if h else self._default_state()
        return EmotionState(primary="neutral", intensity=0.0, trend="decreasing")

    def _default_state(self) -> EmotionState:
        return EmotionState(primary="neutral", intensity=0.5,
                          arousal=0.5, dominance=0.5)
```

### 2.3 ActionParser — 动作提取器

```python
class ActionParser:
    ACTIONS_JSON_PATTERN = re.compile(
        r'\{[^{}]*"actions"\s*:\s*\[[^\]]+\][^{}]*\}',
        re.DOTALL
    )

    def __init__(self):
        self.valid_tags = set(ACTION_LIBRARY.keys())  # 合法标签集合

    def parse(self, raw_output: str) -> list[ActionCommand]:
        """解析 LLM 回复中的动作标签"""
        match = self.ACTIONS_JSON_PATTERN.search(raw_output)
        if not match:
            return [ACTION_LIBRARY["idle"]]

        try:
            raw_tags: list = json.loads(match.group()).get("actions", [])
        except json.JSONDecodeError:
            return [ACTION_LIBRARY["idle"]]

        commands = [
            ACTION_LIBRARY[tag]
            for tag in raw_tags
            if tag in self.valid_tags and tag != "idle"
        ]
        return commands[:2] if commands else [ACTION_LIBRARY["idle"]]
```

```python
import json, os
from pathlib import Path
import datetime

class MemoryManager:
    def __init__(self, base_dir: str):
        self.base = Path(base_dir)
        self.session_dir = self.base / "memory"
        self.session_dir.mkdir(parents=True, exist_ok=True)

        self.relation_file = self.session_dir / "relation.json"
        self.world_file   = self.session_dir / "world_state.json"
        self.audio_dir    = self.session_dir / "audio"
        self.audio_dir.mkdir(exist_ok=True)
        self._init_defaults()

    def _init_defaults(self):
        if not self.relation_file.exists():
            json.dump({"default": {}}, open(self.relation_file, "w", encoding="utf-8"))
        if not self.world_file.exists():
            json.dump({"completed_quests":[],"explored_locations":[],"world_events":[]},
                      open(self.world_file, "w", encoding="utf-8"))

    # ── 短期记忆 ───────────────────────────────
    def save_turn(self, npc_name: str, turn: dict) -> dict:
        sf = self._session_file(npc_name)
        data = json.load(open(sf, encoding="utf-8")) if sf.exists() else {"turns": []}
        data["turns"].append(turn)
        if len(data["turns"]) > MAX_HIST:
            data["turns"] = data["turns"][-MAX_HIST:]
        json.dump(data, open(sf, "w", encoding="utf-8"), ensure_ascii=False)
        return {"relation_change": {npc_name: turn.get("relation_delta", 0)}}

    def get_short_term(self, npc_name: str) -> list[dict]:
        sf = self._session_file(npc_name)
        return json.load(open(sf, encoding="utf-8")).get("turns", []) if sf.exists() else []

    def _session_file(self, npc_name: str) -> Path:
        today = datetime.date.today().isoformat()
        return self.session_dir / f"session_{npc_name}_{today}.json"

    # ── 关系状态 ───────────────────────────────
    def update_relation(self, user_id: str, npc_name: str, delta: int) -> int:
        rel = json.load(open(self.relation_file, encoding="utf-8"))
        rel.setdefault(user_id, {}).setdefault(npc_name, {"intimacy": 50, "last_interact": ""})
        rel[user_id][npc_name]["intimacy"] = clamp(
            rel[user_id][npc_name]["intimacy"] + delta, 0, 100)
        rel[user_id][npc_name]["last_interact"] = now_iso()
        json.dump(rel, open(self.relation_file, "w", encoding="utf-8"), ensure_ascii=False)
        return rel[user_id][npc_name]["intimacy"]

    def get_relation(self, user_id: str, npc_name: str) -> int:
        rel = json.load(open(self.relation_file, encoding="utf-8"))
        return rel.get(user_id, {}).get(npc_name, {}).get("intimacy", 50)

    # ── 长期记忆 ───────────────────────────────
    def update_longterm(self, npc_name: str, events: list[str]):
        ltf = self.session_dir / "longterm" / f"{npc_name}_summary.json"
        data = json.load(open(ltf, encoding="utf-8")) if ltf.exists() \
               else {"npc_name": npc_name, "summary": "", "key_events": []}
        for ev in events:
            if ev not in [e["event"] for e in data["key_events"]]:
                data["key_events"].append({"event": ev, "date": now_iso()})
        data["key_events"] = data["key_events"][-100:]
        data["last_updated"] = now_iso()
        json.dump(data, open(ltf, "w", encoding="utf-8"), ensure_ascii=False)

    def get_longterm(self, npc_name: str) -> dict:
        ltf = self.session_dir / "longterm" / f"{npc_name}_summary.json"
        return json.load(open(ltf, encoding="utf-8")) if ltf.exists() \
               else {"npc_name": npc_name, "summary": "", "key_events": []}
```

### 2.5 AzureTTS — 语音合成（后端）

> ⚠️ 这一段代码只写在这里（Phase 1 后端），Phase 2 Vue 和 Phase 4 Unity 都只播放 `audio_url`，不调用此代码。

```python
import hashlib, os, httpx, asyncio
from pathlib import Path

class AzureTTSService:
    """Azure 神经网络 TTS 合成，输出 .wav 缓存"""

    def __init__(self, audio_dir: Path,
                 api_key: str = os.getenv("AZURE_TTS_KEY", ""),
                 region: str  = os.getenv("AZURE_TTS_REGION", "eastasia")):
        self.audio_dir  = audio_dir
        self.api_key    = api_key
        self.region     = region
        self._token     = None
        self._token_exp = 0.0   # Unix 时间戳

    # ── 缓存策略 ──────────────────────────────
    def _cache_key(self, npc_name: str, text: str, vp: VoiceParams) -> str:
        raw = f"{npc_name}|{text}|{vp.speed}|{vp.pitch}|{vp.volume}|{vp.style}"
        return hashlib.md5(raw.encode()).hexdigest()

    def get_audio_url(self, npc_name: str, text: str, vp: VoiceParams) -> str | None:
        """命中缓存 → 直接返回路径；未命中 → 合成 → 返回路径"""
        if not text or not self.api_key:
            return None  # 无 Key 时静默返回 null，不报错

        cache_key = self._cache_key(npc_name, text, vp)
        cache_file = self.audio_dir / f"{cache_key}.wav"

        if cache_file.exists():
            return f"/audio/{cache_key}.wav"   # 缓存命中

        # 未命中 → 合成
        wav_bytes = asyncio.run(self._synthesize(text, vp))
        if wav_bytes is None:
            return None

        cache_file.write_bytes(wav_bytes)
        return f"/audio/{cache_key}.wav"

    # ── Azure TTS REST API ─────────────────────
    async def _synthesize(self, text: str, vp: VoiceParams) -> bytes | None:
        """调用 Azure TTS，SSML → 返回 WAV 字节"""
        try:
            token = await self._get_token()
        except Exception:
            return None

        ssml = self._build_ssml(text, vp)
        headers = {
            "Authorization": f"Bearer {token}",
            "Content-Type": "application/ssml+xml",
            "X-Microsoft-OutputFormat": "riff-16khz-16bit-mono-pcm",
        }
        url = (f"https://{self.region}.tts.speech.microsoft.com/"
               f"cognitiveservices/v1")

        async with httpx.AsyncClient(timeout=30.0) as client:
            resp = await client.post(url, content=ssml.encode(), headers=headers)

        if resp.status_code != 200:
            print(f"[AzureTTS] 合成失败 {resp.status_code}: {resp.text[:200]}")
            return None
        return resp.content

    async def _get_token(self) -> str:
        """用 API Key 换取 Bearer Token（有效期10分钟）"""
        import time
        if self._token and time.time() < self._token_exp - 60:
            return self._token

        url = f"https://{self.region}.api.cognitive.microsoft.com/sts/v1.0/issueToken"
        headers = {"Ocp-Apim-Subscription-Key": self.api_key}
        async with httpx.AsyncClient(timeout=10.0) as client:
            resp = await client.post(url, headers=headers)
        resp.raise_for_status()
        self._token = resp.text
        self._token_exp = time.time() + 600
        return self._token

    def _build_ssml(self, text: str, vp: VoiceParams) -> str:
        rate_delta = int((vp.speed - 1.0) * 100)
        vol_delta  = int(20 * (vp.volume - 1.0) / 1.0)
        rate_str  = f"{(rate_delta if rate_delta >= 0 else rate_delta):+d}%"
        vol_str   = f"{(vol_delta if vol_delta >= 0 else vol_delta):+d}dB"

        return (
            f"<speak version='1.0' xmlns='http://www.w3.org/2001/10/synthesis'"
            f" xml:lang='zh-CN'>"
            f"<voice name='zh-CN-XiaoxiaoNeural'>"
            f"<prosody rate='{rate_str}' pitch='{vp.pitch}' volume='{vol_str}'>"
            f"{text}"
            f"</prosody></voice></speak>"
        )
```

```python
class JianghuChat:
    def __init__(self):
        # ... V1.0 原有初始化 ...

        )

    def chat(self, user_input: str, history: list,
             npc_name: str, user_id: str = "default") -> dict:

        # 1. 安全过滤
        status, clean = sanitize_input(user_input)
        if status != "ok":
            reply = random.choice(NPC_PROFILES[npc_name]["fallback_replies"][status])
            return self._empty_response(reply)

        # 2. 任务检测（复用 V1.0）
        triggered_quests = self.check_quests(clean, npc_name)

        # 3. 亲密度查询
        relation = self.memory.get_relation(user_id, npc_name)

        # 4. 构建 Prompt（V2.0：含情感+动作规则）
        prompt = ChatPromptTemplate.from_messages([
            SystemMessage(content=(
                profile["system_prompt"]
                + PERMISSION_LOCK.format(npc_name=npc_name)
                + NO_AI_RULES
                + self._build_relation_prompt(npc_name, relation)
                + self._build_emotion_prompt()     # ← Phase 1 新增
                + self._build_action_prompt()      # ← Phase 1 新增
            )),
            HumanMessage(content=self._build_context(
                clean, npc_name, user_id, history, triggered_quests))
        ])

        # 5. LLM 调用
        raw = self.llm.invoke(prompt)

        # 6. 解析 reply
        reply = self._extract_reply(raw)

        # 7. 情感推断
        emotion = self.emotion_engine.infer(reply, npc_name)

        # 8. 动作提取
        actions_cmds = self.action_parser.parse(raw)

        # 9. 语音参数
        voice = VoiceParams.from_emotion(emotion)

        # 10. TTS 合成（后端完成！）
        audio_url = self.azure_tts.get_audio_url(npc_name, reply, voice)

        # 11. 记忆持久化
        rel_delta = RELATION_DELTA.get(emotion.primary, 0)
        if relation < 30 and emotion.primary in ["angry","disgusted","fearful"]:
            rel_delta = max(rel_delta, -2)
        new_relation = self.memory.update_relation(user_id, npc_name, rel_delta)
        memory_update = self.memory.save_turn(npc_name, {
            "user": clean,
            "assistant": reply,
            "emotion": emotion.primary,
            "intensity": emotion.intensity,
            "relation_delta": rel_delta,
            "timestamp": now_iso(),
        })

        # 12. 任务持久化
        for q in triggered_quests:
            if not any(x.get("qid") == q["qid"] for x in self.active_quests):
                self.active_quests.append(q)

        return {
            "reply": reply,
            "emotion_state": emotion.to_dict(),
            "voice_params": voice.to_dict(),
            "audio_url": audio_url,                        # ← Phase 1 新增
            "actions": [a.tag for a in actions_cmds],      # ← Phase 1 新增
            "new_quests": triggered_quests,
            "memory_update": memory_update,
        }
```

```csharp
public class ChatClient : MonoBehaviour
{
    private const string BASE_URL = "http://127.0.0.1:8000";

    public async Task<NPCResponsePacket> SendAsync(
        string message, string npcName, string userId = "default")
    {
        var payload = new {
            message,
            npc_name = npcName,
            user_id  = userId
        };

        string json = JsonUtility.ToJson(payload);
        byte[] body = System.Text.Encoding.UTF8.GetBytes(json);

        var request = new UnityWebRequest(BASE_URL + "/api/chat", "POST");
        request.uploadHandler   = new UploadHandlerRaw(body);
        request.downloadHandler = new DownloadHandlerBuffer();
        request.SetRequestHeader("Content-Type", "application/json");
        request.timeout = 10;

        var operation = request.SendWebRequest();
        float elapsed = 0f;
        while (!operation.isDone && elapsed < 10f) {
            elapsed += Time.deltaTime;
            await Task.Yield();
        }

        if (request.result != UnityWebRequest.Result.Success) {
            Debug.LogError($"[ChatClient] 请求失败: {request.error}");
            return new NPCResponsePacket {
                reply = "（网络不畅，请稍后再试）",
                emotion_state = new EmotionState(),
                actions = new List<string> { "idle" }
            };
        }

        return JsonUtility.FromJson<NPCResponsePacket>(
            request.downloadHandler.text);
    }
}
```

```csharp
public class AudioPlayer : MonoBehaviour
{
    [SerializeField] private AudioSource audioSource;
    private const string BASE_URL = "http://127.0.0.1:8000";

    /// <summary>
    /// 从后端 audio_url 下载并播放 .wav。
    /// audio_url 为 null 时静默跳过。
    /// </summary>
    public async Task PlayURL(string audioUrl)
    {
        if (string.IsNullOrEmpty(audioUrl)) return;

        string url = BASE_URL + audioUrl;

        using (var request = UnityWebRequest.Get(url))
        {
            await request.SendWebRequest();

            if (request.result != UnityWebRequest.Result.Success) {
                Debug.LogWarning($"[AudioPlayer] 下载失败: {request.error}");
                return;
            }

            var clip = DownloadHandlerAudioClip.GetContent(request);
            audioSource.clip = clip;
            audioSource.Play();

            // 等待播放完成
            await new WaitUntil(() => !audioSource.isPlaying);
        }
    }

    /// <summary>停止当前音频</summary>
    public void Stop()
    {
        if (audioSource.isPlaying) audioSource.Stop();
    }
}
```

### 5.4 EmotionDriver — 表情驱动

```csharp
public class EmotionDriver : MonoBehaviour
{
    [SerializeField] private SkinnedMeshRenderer faceMesh;

    // 各 NPC 独立映射（通过 Inspector 赋值，JSON 或直接填）
    [SerializeField] private List<EmotionBlendMapping> emotionMappings;

    [System.Serializable]
    public class EmotionBlendMapping {
        public string emotion;               // "excited"
        public List<BlendEntry> blends;       // Blendshape 映射
    }
    [System.Serializable]
    public class BlendEntry {
        public string blendName = "";         // "BS_Smile"
        public float baseWeight = 1.0f;       // 基础权重（乘 intensity）
    }

    private Coroutine activeRoutine;

    public void ApplyEmotion(EmotionState state)
    {
        if (activeRoutine != null) StopCoroutine(activeRoutine);
        activeRoutine = StartCoroutine(LerpEmotion(state));
    }

    IEnumerator LerpEmotion(EmotionState state)
    {
        var mapping = emotionMappings
            .FirstOrDefault(m => m.emotion == state.primary)
            ?? emotionMappings.FirstOrDefault(); // neutral 兜底

        foreach (var entry in mapping.blends)
        {
            int idx = faceMesh.sharedMesh.GetBlendShapeIndex(entry.blendName);
            if (idx < 0) continue;
            float target = entry.baseWeight * state.intensity;
            StartCoroutine(LerpBlend(idx, target, 0.3f));
        }
        yield break;
    }

    IEnumerator LerpBlend(int shapeIndex, float targetWeight, float duration)
    {
        float start = faceMesh.GetBlendShapeWeight(shapeIndex);
        float t = 0f;
        while (t < duration)
        {
            t += Time.deltaTime;
            faceMesh.SetBlendShapeWeight(shapeIndex,
                Mathf.Lerp(start, targetWeight, t / duration));
            yield return null;
        }
        faceMesh.SetBlendShapeWeight(shapeIndex, targetWeight);
    }

    /// <summary>表情衰减到 neutral</summary>
    public void Decay()
    {
        for (int i = 0; i < faceMesh.sharedMesh.blendShapeCount; i++)
            StartCoroutine(LerpBlend(i, 0f, 0.5f));
    }
}
```

### 5.5 ActionController — 动作驱动

```csharp
public class ActionController : MonoBehaviour
{
    [SerializeField] private Animator animator;
    private const int BASE_LAYER   = 0;
    private const int ACTION_LAYER = 1;

    private Queue<string> pendingActions = new();
    private bool isExecuting = false;

    public void EnqueueActions(List<string> actionTags)
    {
        foreach (var tag in actionTags.Take(2))
            pendingActions.Enqueue(tag);
        if (!isExecuting) ProcessNext();
    }

    void ProcessNext()
    {
        if (pendingActions.Count == 0) { ReturnToIdle(); return; }
        isExecuting = true;
        ExecuteAction(pendingActions.Dequeue());
    }

    void ExecuteAction(string tag)
    {
        animator.SetLayerWeight(ACTION_LAYER, 1f);
        animator.CrossFade(tag, 0.3f, ACTION_LAYER);

        // 获取动作时长（简单用 WaitForSeconds，或从 AnimationClip 读）
        StartCoroutine(FinishAndContinue(tag, 1.5f));
    }

    IEnumerator FinishAndContinue(string tag, float duration)
    {
        yield return new WaitForSeconds(duration);
        ReturnToIdle();
        yield return new WaitForSeconds(0.3f);
        ProcessNext();
    }

    void ReturnToIdle()
    {
        animator.SetLayerWeight(ACTION_LAYER, 0f);
        animator.CrossFade("Idle", 0.3f, BASE_LAYER);
    }
}
```

### 5.6 NPCController — 状态机主控

```csharp
public class NPCController : MonoBehaviour
{
    public enum State { Idle, PreProcess, Speaking }

    [SerializeField] private string npcName = "店小二";
    [SerializeField] private string userId   = "default";

    [Header("子系统")]
    [SerializeField] private ChatClient       chatClient;
    [SerializeField] private EmotionDriver    emotionDriver;
    [SerializeField] private ActionController  actionController;
    [SerializeField] private AudioPlayer      audioPlayer;     // ✅ 纯播放
    [SerializeField] private DialogueUI        dialogueUI;

    private State currentState = State.Idle;

    public async void OnUserSendMessage(string message)
    {
        if (currentState != State.Idle) return;
        currentState = State.PreProcess;

        dialogueUI.AppendUser(message);
        dialogueUI.SetLoading(true);

        // 1. 发请求
        var packet = await chatClient.SendAsync(message, npcName, userId);
        dialogueUI.SetLoading(false);

        // 2. 显示回复文本
        dialogueUI.AppendNPC(packet.reply);

        // 3. 表情/动作并行先行（~300ms）
        emotionDriver.ApplyEmotion(packet.emotion_state);
        if (packet.actions?.Count > 0)
            actionController.EnqueueActions(packet.actions);

        await Task.Delay(300);   // 先行等待

        // 4. 音频播放（只调 audio_url，不发 TTS！）
        currentState = State.Speaking;
        if (!string.IsNullOrEmpty(packet.audio_url))
            await audioPlayer.PlayURL(packet.audio_url);

        // 5. 收尾：动作归零 + 表情衰减
        emotionDriver.Decay();
        currentState = State.Idle;
    }
}
```
